var arr[];
function addFeedback(){
 //Fill the required logic  
 var feed=document.getElementById("feedback");
 var result=document.getElementById("result");
 arr.push(feed.value);
 feed.value=" ";
 result.innerHTML="<h2>Feedback Details</h2><br>Successfully Added Feedback Details!";
}

function displayFeedback(){
    //Fill the required logic
    var n=1;
    var res="<h2>Feedback Details</h2>";
    for(var i in arr){
        res+="Feedback "+n.toString() +"<br>" +arr[i] +"<br>";
        n++;
    }
    document.getElementById("result").innerHTML=res;

arr=[];
    
}
